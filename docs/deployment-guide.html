1. System Architecture Overview

CLOUD (SaaS) — Recommended

Fully managed on Vercel (frontend) + AWS Elastic Beanstalk (backend) + MongoDB Atlas.

  • Zero infrastructure to manage
  • Auto-scaling, CDN, SSL included
  • Deployment in <1 hour
  • SOC 2 Type II compliant hosting

ON-PREMISE — Available

Self-hosted using Docker Compose on customer's infrastructure.

  • Full data sovereignty
  • Air-gapped deployment possible
  • Custom SSL/network policies
  • Requires IT operations team

2.2 Step-by-Step Setup

1
Create MongoDB Atlas Cluster
Sign up at mongodb.com/atlas → Create cluster → Get connection string:
mongodb+srv://user:pass@cluster.mongodb.net/hawkeye
2
Create AWS S3 Bucket
aws s3 mb s3://hawkeye-uploads-prod → Create IAM user with S3 access → Note Access Key ID + Secret
3
Deploy Backend to Vercel
cd backend
npm install -g vercel
vercel login
vercel --prod
Set environment variables in Vercel dashboard:
MONGO_URI=mongodb+srv://...
JWT_SECRET=your-secret-key-here
NODE_ENV=production
AWS_ACCESS_KEY_ID=AKIA...
AWS_SECRET_ACCESS_KEY=...
AWS_S3_BUCKET=hawkeye-uploads-prod
AWS_REGION=us-east-1
4
Deploy Frontend to Vercel
cd frontend
vercel --prod
Set environment variables:
APP_API_BASE_URL=https://your-backend.vercel.app
NEXT_PUBLIC_APP_API_BASE_URL=https://your-backend.vercel.app
NEXT_SESSION_SECRET=same-as-JWT_SECRET
5
Configure Custom Domain (Optional)
Vercel dashboard → Domains → Add app.yourcompany.com → Update DNS CNAME to cname.vercel-dns.com
6
Create First Admin User
curl -X POST https://your-backend.vercel.app/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{"email":"admin@company.com","password":"SecureP@ss123",
       "role":"superadmin","firstName":"Admin","lastName":"User"}'
7
Configure Modules
Login as admin → /admin/module-config → Enable modules per customer type → Set vocabulary → Set industry profile
Cloud deployment complete. Total time: ~30 minutes. No servers to manage.

3.2 Docker Compose Setup

# docker-compose.yml

version: '3.8'
services:
  # ── MongoDB ──────────────────────────────────────────────
  mongo:
    image: mongo:7
    restart: always
    volumes:
      - mongo_data:/data/db
    environment:
      MONGO_INITDB_ROOT_USERNAME: hawkeye
      MONGO_INITDB_ROOT_PASSWORD: ${MONGO_PASSWORD}
    ports:
      - "27017:27017"

  # ── MinIO (S3-compatible storage) ────────────────────────
  minio:
    image: minio/minio:latest
    restart: always
    command: server /data --console-address ":9001"
    volumes:
      - minio_data:/data
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: ${MINIO_PASSWORD}
    ports:
      - "9000:9000"
      - "9001:9001"

  # ── Backend API ──────────────────────────────────────────
  backend:
    build: ./backend
    restart: always
    depends_on: [mongo, minio]
    environment:
      NODE_ENV: production
      PORT: 8102
      MONGO_URI: mongodb://hawkeye:${MONGO_PASSWORD}@mongo:27017/hawkeye?authSource=admin
      JWT_SECRET: ${JWT_SECRET}
      AWS_ACCESS_KEY_ID: minioadmin
      AWS_SECRET_ACCESS_KEY: ${MINIO_PASSWORD}
      AWS_S3_BUCKET: hawkeye-uploads
      AWS_REGION: us-east-1
      S3_ENDPOINT: http://minio:9000
      FE_BASE_URL: https://${DOMAIN}
    ports:
      - "8102:8102"

  # ── Frontend (Next.js) ──────────────────────────────────
  frontend:
    build: ./frontend
    restart: always
    depends_on: [backend]
    environment:
      APP_API_BASE_URL: http://backend:8102
      NEXT_PUBLIC_APP_API_BASE_URL: https://${DOMAIN}/api
      NEXT_SESSION_SECRET: ${JWT_SECRET}
    ports:
      - "3000:3000"

  # ── Nginx (reverse proxy + SSL) ─────────────────────────
  nginx:
    image: nginx:alpine
    restart: always
    depends_on: [frontend, backend]
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - ./ssl:/etc/nginx/ssl:ro
    ports:
      - "80:80"
      - "443:443"

volumes:
  mongo_data:
  minio_data:

3.4 Nginx Configuration

# nginx.conf
events { worker_connections 1024; }
http {
  upstream frontend { server frontend:3000; }
  upstream backend  { server backend:8102; }

  server {
    listen 443 ssl;
    server_name hawkeye.yourcompany.com;

    ssl_certificate     /etc/nginx/ssl/cert.pem;
    ssl_certificate_key /etc/nginx/ssl/key.pem;

    # Frontend (all routes except /api)
    location / {
      proxy_pass http://frontend;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
    }

    # Backend API
    location /api {
      proxy_pass http://backend;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
    }

    # WebSocket (Socket.IO)
    location /socket.io {
      proxy_pass http://backend;
      proxy_http_version 1.1;
      proxy_set_header Upgrade $http_upgrade;
      proxy_set_header Connection "upgrade";
    }

    # Health check
    location /health {
      proxy_pass http://backend;
    }
  }

  server {
    listen 80;
    return 301 https://$host$request_uri;
  }
}

6. Backup & Disaster Recovery

Cloud (Atlas)

MongoDB Atlas provides automatic daily backups with point-in-time recovery. Configure in Atlas dashboard → Backup → Continuous Backup.

On-Premise

# Daily backup cron job
0 2 * * * docker exec hawkeye-mongo-1 mongodump \
  --uri="mongodb://hawkeye:$MONGO_PASSWORD@localhost:27017/hawkeye?authSource=admin" \
  --out=/backup/$(date +\%Y-\%m-\%d) \
  && tar czf /backup/hawkeye-$(date +\%Y-\%m-\%d).tar.gz /backup/$(date +\%Y-\%m-\%d) \
  && rm -rf /backup/$(date +\%Y-\%m-\%d)

# Restore from backup
docker exec -i hawkeye-mongo-1 mongorestore \
  --uri="mongodb://hawkeye:$MONGO_PASSWORD@localhost:27017/hawkeye?authSource=admin" \
  --drop /backup/2026-04-17/
